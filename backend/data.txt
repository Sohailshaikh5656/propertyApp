SELECT 
    mp.date,
    DAYNAME(mp.date) AS day, 
    m.name AS meal,
    od.qty,
    (WEEK(mp.date, 1) - WEEK(DATE_SUB(mp.date, INTERVAL DAYOFMONTH(mp.date)-1 DAY), 1) + 1) AS week_number
FROM tbl_meal_provider AS mp
JOIN tbl_meal AS m ON mp.meal_id = m.id
JOIN tbl_order_details AS od ON od.meal_provider_id = mp.id
WHERE YEAR(mp.date) = YEAR(CURRENT_DATE) 
AND MONTH(mp.date) = MONTH(CURRENT_DATE)
ORDER BY week_number, mp.date;
